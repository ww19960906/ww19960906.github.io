# 添加当前目录的所有文件到暂存区{% highlight ruby %}$ git add .{% endhighlight %}

# 提交暂存区到仓库区{% highlight ruby %}$ git commit -m [message]{% endhighlight %}

# 提交暂存区的指定文件到仓库区{% highlight ruby %}$ git commit [file1] [file2] ... -m [message]{% endhighlight %}

git 推送

# 下载远程仓库的所有变动{% highlight ruby %}$ git fetch [remote]{% endhighlight %}

# 显示所有远程仓库{% highlight ruby %}$ git remote -v{% endhighlight %}

# 显示某个远程仓库的信息{% highlight ruby %}$ git remote show [remote]{% endhighlight %}

# 增加一个新的远程仓库，并命名{% highlight ruby %}$ git remote add [shortname] [url]{% endhighlight %}

# 推送所有分支到远程仓库{% highlight ruby %}$ git push [remote] --all{% endhighlight %}
